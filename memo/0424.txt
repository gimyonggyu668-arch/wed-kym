create table tb1_student(
id int primary key,
name varchar(100),
major varchar(100),
gender char(1) not null default 'W',
constraint ban_char check(gender in ('M', 'W')),
constraint ban_date check(birth >= '1980-01-01')
)
# default : 컬럼에 값이 아예 주어지지 않았을 때 자동으로 들어가는
# not null : 컬럼에 null 값 자체를 허용하지 않겠다

# 같이 사용하면 값이 생략시 'W' 가 들어가게 되고, 누군가 null을 
# 넣으려고 하면 DB가 거부한다.

#not null 이 없고 default 만 있을 때 누군가 gender 자리에
#null라고 넣으면 데이터가 null이라고 들어가게 된다.

프로그램과 운영체제 또는 여러 시스템 사이에서 서로 연결하고 도와주는

미들웨어는 사용자가 직접 눈으로 보는 프로그램은 아니지만, 시스템이 정상적으로 동작하기 위해 뒤에서 중요한 역할을 한다.

미들웨어를 사용하는 이유 
"운영체제랑 프로그램을 연결"

1. 서로 다른 시스템을 연결하기 위해 
- 시스템 마다 언어, 데이터 형식, 통신 방식이 다를 수 있다.
이러한 시스템들을 연결해주는 역할을 한다.

2. 개발자가 공통 기능을 직접 만들지 않아도  된다.
로그인 세션 관리, 보안처리, 데이터 변환, 메세지 전송
이런 기능을 개발자 전부 직접 만들면 시간이 오래걸리고 실수도 많아진다.
미들웨어는 공통 기능을 제공해준다.

SQL문장의 종류

DDL(Data Definintion Language) :  데이터 정의어
- CREATE: 테이블이나 인덱스, 뷰 등 데이터베이스 객체를 생성한다.
- DROP: 생성된 데이터 베이스 객체를 영구히 삭제
- ALTER: 이미 생성된 데이터베이스 객체를 수정
- TRUNCATE: 테이블의 데이터를 통째로 삭제
 
DML(Data Manipulation Language) : 데이터 조작어 
DCL(Data Controll Language) :  데이터 제어어
TCL(Transaction Controll Language) : 트랜잭션 제어어

SQL문장의 종류



DDL(Data Definintion Language) :  데이터 정의어


- CREATE: 테이블이나 인덱스, 뷰 등 데이터베이스 객체를 생성한다.
- DROP: 생성된 데이터 베이스 객체를 영구히 삭제
- ALTER: 이미 생성된 데이터베이스 객체를 수정
- TRUNCATE: 테이블의 데이터를 통째로 삭제

DML(Data Manipulation Language) : 데이터 조작어
DCL(Data Controll Language) :  데이터 제어어
TCL(Transaction Controll Language) : 트랜잭션 제어어
